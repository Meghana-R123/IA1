3️⃣ Linear Search
LinearSearch.java
package lab;

public class LinearSearch {
    public int search(int[] arr, int key) {
        for (int i = 0; i < arr.length; i++) {
            if (arr[i] == key)
                return i;
        }
        return -1;
    }
}
LinearSearchTest.java
package lab;

import static org.junit.Assert.*;
import org.junit.Test;

public class LinearSearchTest {

    LinearSearch ls = new LinearSearch();

    @Test
    public void testSearch() {
        int[] arr = {1, 2, 3, 4};
        assertEquals(2, ls.search(arr, 3));
    }
}

5️⃣ Bank Account
BankAccount.java
package lab;

public class BankAccount {
    double balance = 0;

    public void deposit(double amount) {
        balance += amount;
    }

    public void withdraw(double amount) {
        balance -= amount;
    }

    public double getBalance() {
        return balance;
    }
}
BankAccountTest.java
package lab;

import static org.junit.Assert.*;
import org.junit.Test;

public class BankAccountTest {

    @Test
    public void testDeposit() {
        BankAccount acc = new BankAccount();
        acc.deposit(100);
        assertEquals(100, acc.getBalance(), 0);
    }

    @Test
    public void testWithdraw() {
        BankAccount acc = new BankAccount();
        acc.deposit(100);
        acc.withdraw(40);
        assertEquals(60, acc.getBalance(), 0);
    }
}

________________________________________
📁 Step 1: Prepare GitHub Repo
Upload these files:

        System.out.println("Docker + Jenkins Working");
    }
}
________________________________________
🔹 Dockerfile
FROM eclipse-temurin:17-jdk
WORKDIR /app
COPY . .
RUN javac Grade.java
CMD ["java", "Grade"]
________________________________________
⚙️ Step 2: Create Jenkins Job
•	Open Jenkins → http://localhost:8080 
•	Click New Item 
•	Enter name: Docker-Jenkins 
•	Select Pipeline → OK 
________________________________________
📄 Step 3: Add Pipeline Script
pipeline {
    agent any
    stages {
        stage('Clone Code') {
            steps {
                git 'https://github.com/your-username/your-repo.git'
            }
        }
        stage('Build Docker Image') {
            steps {
                bat 'docker build -t my-app .'
            }
        }
        stage('Run Container') {
            steps {
                bat 'docker run -d my-app'
            }
        }
    }
}

5. Save and apply 

6. Once build is success check the image created in docker desktop
